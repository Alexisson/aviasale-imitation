CREATE VIEW DEP_ECON AS
SELECT FLIGHT.Y_PRICE, AIRLINE.AIRLINE, AIRLINE.IATA, FLIGHT.FLIGHT_NUM, DEP_SCHD.SCHD_TIME, 
AIRPORT.CITY AS DEP_CITY, AIRPORT.AIRPORT AS DEP_AP  
FROM AIRLINE INNER JOIN FLIGHT ON AIRLINE.ID_AIRLINE = FLIGHT.AIRLINE_ID_AIRLINE
INNER JOIN DEP_SCHD ON FLIGHT.ID_FLIGHT = DEP_SCHD.FLIGHT_ID_FLIGHT
INNER JOIN AIRPORT ON DEP_SCHD.AIRPORT_ID_AIRPORT = AIRPORT.ID_AIRPORT;

CREATE VIEW DEP_BUSINESS AS
SELECT FLIGHT.C_PRICE, AIRLINE.AIRLINE, AIRLINE.IATA, FLIGHT.FLIGHT_NUM, DEP_SCHD.SCHD_TIME, 
AIRPORT.CITY AS DEP_CITY, AIRPORT.AIRPORT AS DEP_AP  
FROM AIRLINE INNER JOIN FLIGHT ON AIRLINE.ID_AIRLINE = FLIGHT.AIRLINE_ID_AIRLINE
INNER JOIN DEP_SCHD ON FLIGHT.ID_FLIGHT = DEP_SCHD.FLIGHT_ID_FLIGHT
INNER JOIN AIRPORT ON DEP_SCHD.AIRPORT_ID_AIRPORT = AIRPORT.ID_AIRPORT;

CREATE VIEW DEP AS
SELECT AIRLINE.AIRLINE, AIRLINE.IATA, FLIGHT.FLIGHT_NUM, DEP_SCHD.SCHD_TIME, 
AIRPORT.CITY AS DEP_CITY, AIRPORT.AIRPORT AS DEP_AP  
FROM AIRLINE INNER JOIN FLIGHT ON AIRLINE.ID_AIRLINE = FLIGHT.AIRLINE_ID_AIRLINE
INNER JOIN DEP_SCHD ON FLIGHT.ID_FLIGHT = DEP_SCHD.FLIGHT_ID_FLIGHT
INNER JOIN AIRPORT ON DEP_SCHD.AIRPORT_ID_AIRPORT = AIRPORT.ID_AIRPORT;

CREATE VIEW ARR AS
SELECT FLIGHT.FLIGHT_NUM, ARR_SCHD.SCHD_TIME, 
AIRPORT.CITY AS ARR_CITY, AIRPORT.AIRPORT AS ARR_AP  
FROM AIRLINE INNER JOIN FLIGHT ON AIRLINE.ID_AIRLINE = FLIGHT.AIRLINE_ID_AIRLINE
INNER JOIN ARR_SCHD ON FLIGHT.ID_FLIGHT = ARR_SCHD.FLIGHT_ID_FLIGHT
INNER JOIN AIRPORT ON ARR_SCHD.AIRPORT_ID_AIRPORT = AIRPORT.ID_AIRPORT;
